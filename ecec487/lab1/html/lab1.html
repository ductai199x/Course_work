Problem 2

mu = [0,1]
sigma = [0.1 0; 0 0.1]


[X1,X2] = meshgrid(linspace(-5,5,100)',linspace(-5,5,100)');
X = [X1(:) X2(:)];
p = mvnpdf(X, mu, sigma);
surf(X1,X2,reshape(p,100,100));

mu =

     0     1


sigma =

    0.1000         0
         0    0.1000
